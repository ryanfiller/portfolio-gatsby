function glitchAnimation() {
 
    var array = []
    
    for (var i = 0; i <= 20; i++) { 
 
        var random1 = Math.floor(Math.random() * 100) + 1
        var random2 = Math.floor(Math.random() * 100) + 1
 
        array.push(`
            ${i * 5}% {
                clip: rect(${random1}px,9999px,${random2}px,0);
            }
        `)
    }
 
    return array.join("");
};
